[docs]def clahe_equal(grayimg):
    """
    Applies contrast limited adaptive histogram equalisation to the image.

    Image divided into 8x8 tiles and each tile is histogram equalised.
    Contrast limiting is applied to prevent any noise being amplified.

    :param numpy.ndarray grayimg: The grayscale image.
    :return: clah equalised image
    :rtype: numpy.ndarray
    """
    assert len(grayimg.shape) == 2  # Ensure 2D array (gray scale image)
    # create clahe object
    clahe = cv2.createCLAHE(clipLimit=2.0, tileGridSize=(8, 8))
    return clahe.apply(grayimg)




[docs]def normalize(grayimg):
    """
    Normalise/standardise all pixels.

    :param numpy.ndarray grayimg: The grayscale image.
    :return: clah equalised image
    :rtype: numpy.ndarray
    """
    assert len(grayimg.shape) == 2  # Ensure 2D array (gray scale image)
    norm = np.empty(grayimg.shape)
    norm = ((grayimg - np.mean(grayimg)) / np.std(grayimg))
    return (((norm - np.min(norm)) /
             (np.max(norm) - np.min(norm))) * 255).astype("uint8")




[docs]def adjust_gamma(grayimg, gamma=1.0):
    """
    Adjust gamma of the image.

    :param numpy.ndarray grayimg: The grayscale image.
    :return: image with gamma adjusted
    :rtype: numpy.ndarray
    """
    assert len(grayimg.shape) == 2  # Ensure 2D array (gray scale image)
    # Create look up table mapping pixels [0,255] to their adjusted values
    invgamma = 1.0 / gamma
    newimg = np.empty(grayimg.shape)
    table = np.array([((i / 255.0) ** invgamma) * 255
                      for i in np.arange(0, 256)]).astype("uint8")
    newimg = cv2.LUT(grayimg, table)
    return newimg
